SELECT      s.specObjID,   s.ra, s.dec   into MyDB.hst_qsos from MyDB.hst_radec AS m   OUTER APPLY dbo.fGetNearestSpecObjAllEq( m.RA, m.DEC, #) AS n   LEFT JOIN SpecObjAll AS s ON n.specObjID = s.specObjID
